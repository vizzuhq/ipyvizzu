    Info - How to setup Chart 
    import pandas as pd
from ipyvizzu import Chart, Data, Config, Style

df = pd.read_csv(
    "https://ipyvizzu.vizzuhq.com/0.17/assets/data/chart_types_eu_data_6.csv",
    dtype={"Year": str, "Timeseries": str},
)
data = Data()
data.add_df(df)

chart = Chart()
chart.animate(data)
    • add a new Dimension to the Color channel
    chart.animate(
    Config(
        {
            "channels": {"x": "Year", "y": "Value 2 (+)"},
            "geometry": "line",
            "coordSystem": "polar",
        }
    )
)

chart.animate(Config({"channels": {"color": "Country"}}))

chart.feature("tooltip", True)
