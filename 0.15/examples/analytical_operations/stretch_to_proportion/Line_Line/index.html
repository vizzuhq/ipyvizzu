    Info - How to setup Chart 
    import pandas as pd
from ipyvizzu import Chart, Data, Config, Style

data_frame = pd.read_csv(
    "https://ipyvizzu.vizzuhq.com/0.15/assets/data/chart_types_eu_data_8.csv",
    dtype={"Year": str, "Timeseries": str},
)
data = Data()
data.add_data_frame(data_frame)

chart = Chart()
chart.animate(data)
    • add the Dimension which is on the Color channel to the Y-axis too
    • switch the Geometry from Line to Area
    • set the Align parameter to 'stretch'
    chart.animate(
    Config(
        {
            "channels": {
                "x": "Year",
                "y": "Value 2 (+)",
                "color": "Country",
            },
            "geometry": "line",
        }
    )
)

chart.animate(
    Config(
        {
            "channels": {"y": ["Value 2 (+)", "Country"]},
            "geometry": "area",
            "align": "stretch",
        }
    )
)

chart.feature("tooltip", True)
